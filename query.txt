SELECT * FROM cours
INNER JOIN intervenant ON cours.intervenant_cours = intervenant.id_intervenant  WHERE promotion = (SELECT promo_etudiante FROM `etudiant` WHERE id_etudiant = 1) and Date = CURDATE()
